\section{Introduction}
%==============================================================================

OghmaNotes is a web-based learning platform for students to take markdown notes, upload and extract text from PDFs, search their materials intelligently, and ask AI questions with answers backed by their own course materials (Retrieval-Augmented Generation).

\subsection{Purpose}

This document defines the realistic scope, architecture, and features for OghmaNotes. It reflects actual codebase capabilities and planned development phases.

\textbf{Important Note:} ``In Scope'' means features are planned for the project. Not all are implemented yet. See each phase section and the implementation status notes for actual progress.

\subsection{Audience}

For: Development team, supervisors, stakeholders.

\section{Scope}
%==============================================================================

\subsection{In Scope}

\subsubsection{Core Learning System}

\begin{itemize}
    \item \textbf{Note Taking:} Markdown editor with offline support (PWA), auto-save, conflict resolution
    \item \textbf{Note Organization:} Hierarchical folder structure with drag-and-drop support
    \item \textbf{File Storage:} AWS S3 for notes and files
    \item \textbf{PDF Upload:} Store PDFs up to 100MB per file
\end{itemize}

\subsubsection{Search System}

\begin{itemize}
    \item \textbf{Fuzzy Keyword Search:} Fast full-text search across notes and PDFs (\textless5ms)
    \item \textbf{Semantic Search:} Vector-based similarity search using embeddings (\textless50ms)
    \item \textbf{Search UI:} Command palette (Cmd+K) with filters and snippets
    \item \textbf{Note Extraction:} Text extraction from PDFs with metadata preservation
\end{itemize}

\subsubsection{AI-Powered Features}

\begin{itemize}
    \item \textbf{RAG Chat:} Ask questions about your notes and PDFs, receive AI answers with citations
    \item \textbf{PDF Chunking:} Automatic text segmentation (500 tokens per chunk, 50--100 token overlap)
    \item \textbf{Vector Embeddings:} 1536-dimensional embeddings for semantic search
    \item \textbf{Streaming Responses:} Token-by-token response generation
    \item \textbf{Source Citations:} Clickable links to source PDF locations and notes
\end{itemize}

\subsubsection{Learning Tools}

\begin{itemize}
    \item \textbf{Quiz Generation:} Auto-generate quizzes from notes and PDFs
    \item \textbf{Quiz Types:} Multiple choice, short answer, true/false
    \item \textbf{Difficulty Levels:} Easy, medium, hard
    \item \textbf{Flashcard System:} Spaced repetition using SM-2 algorithm
    \item \textbf{Flashcard Deck Management:} Create, review, and track progress
    \item \textbf{Analytics (Basics):} Overall quiz scores, flashcard mastery, study time tracking (Phase 3, Weeks 8--12)
\end{itemize}

\subsubsection{Canvas LMS Integration (Phase 3)}

\textit{Status: Planned for Phase 3 (Weeks 8--12). Zero code currently implemented.}

\begin{itemize}
    \item \textbf{OAuth Connection:} Secure connection to Canvas LMS
    \item \textbf{Assignment Sync:} Auto-import assignments and due dates
    \item \textbf{Calendar Mapping:} Display Canvas due dates on assignment calendar
    \item \textbf{Daily Sync:} Automatic sync of new assignments (with manual refresh option)
\end{itemize}

\subsubsection{Calendar \& Assignment Tracking}

\begin{itemize}
    \item \textbf{Assignment Calendar:} View Canvas assignments and due dates
    \item \textbf{Due Date Display:} Clear visualization of upcoming deadlines
    \item \textbf{Integration with Study Tools:} Link quizzes and flashcards to assignments
\end{itemize}

\subsubsection{Authentication \& Security}

\begin{itemize}
    \item Email and password registration
    \item Email verification required
    \item JWT authentication (24-hour expiry)
    \item Secure password reset (1-hour token, single-use)
    \item Passwords hashed with bcrypt (10+ rounds)
    \item Account deletion with soft-delete grace period (7 days)
    \item Settings management (email, password, preferences)
\end{itemize}

\subsubsection{User Interface}

\begin{itemize}

    \item Responsive design (desktop, tablet)
    \item Real-time form validation
    \item Keyboard-first navigation
    \item Actionable error messages
    \item Markdown editor with live preview
\end{itemize}

\section{Functional Requirements}
%==============================================================================

\subsection{User Authentication}

\begin{enumerate}
    \item System shall allow users to register with email and password
    \item System shall validate email addresses and send verification emails
    \item System shall require email verification before login
    \item System shall allow users to login with verified email and password
    \item System shall issue JWT tokens valid for 24 hours
    \item System shall support secure password reset via email tokens (1-hour validity)
    \item System shall hash passwords using bcrypt with minimum 10 rounds
    \item System shall lock accounts after 5 failed login attempts
    \item System shall support user logout
    \item System shall allow users to delete accounts (soft-delete with 7-day recovery window)
\end{enumerate}

\subsection{Note Taking}

\begin{enumerate}
    \item System shall allow authenticated users to create, read, update, and delete notes
    \item System shall store notes as Markdown with metadata (title, created\_at, updated\_at)
    \item System shall support note hierarchy (parent-child relationships)
    \item System shall enable folder-like organization with drag-and-drop reordering
    \item System shall auto-save notes to local storage (IndexedDB) every 5 seconds
    \item System shall sync notes to S3 on reconnection
    \item System shall implement conflict resolution (last-write-wins)
    \item System shall support pinning and soft-deleting notes
    \item System shall preserve note creation and modification timestamps
\end{enumerate}

\subsection{PDF Upload \& Storage}

\begin{enumerate}
    \item System shall allow authenticated users to upload PDF files up to 100MB
    \item System shall validate file format and size before processing
    \item System shall store PDFs in S3 with secure access
    \item System shall generate presigned URLs for authenticated downloads (1-hour expiry)
    \item System shall extract text from uploaded PDFs
    \item System shall preserve PDF metadata (file name, upload date, page count)
    \item System shall provide upload progress indication
    \item System shall handle upload failures gracefully with error messages
\end{enumerate}

\subsection{Search}

\begin{enumerate}
    \item System shall provide fuzzy keyword search across all notes and PDFs
    \item Keyword search shall complete in under 5 milliseconds
    \item System shall provide semantic vector search using embeddings
    \item Semantic search shall complete in under 50 milliseconds
    \item System shall support search filters (by tags, document type, date range)
    \item System shall display search results with relevance scores and text snippets
    \item System shall implement command palette (Cmd+K) for keyboard-first search
    \item System shall rank results by relevance with pagination support
\end{enumerate}

\subsection{RAG Chat \& AI Features}

\begin{enumerate}
    \item System shall extract text from PDFs and split into semantic chunks (500 tokens, 50--100 overlap)
    \item System shall generate 1536-dimensional embeddings for text chunks
    \item System shall store embeddings in PostgreSQL with pgvector extension
    \item System shall accept user questions in free text
    \item System shall retrieve relevant document chunks using hybrid search
    \item System shall generate LLM responses with citations to source materials
    \item System shall stream responses token-by-token to the user
    \item System shall include clickable citations linking to original PDF locations
    \item System shall handle LLM API failures gracefully
    \item System shall support user API key management for LLM providers
\end{enumerate}

\section{Design Decisions}
%==============================================================================

\subsection{User API Key Management}

\textbf{Decision:} Users provide their own LLM API keys (OpenAI, Anthropic, etc).

\textbf{Rationale:}
\begin{itemize}
    \item Eliminates backend LLM infrastructure and costs
    \item Users maintain control of their API keys and usage
    \item Reduces vendor lock-in
    \item Simplifies deployment and scaling
\end{itemize}

\textbf{Implementation:}
\begin{itemize}
    \item API keys stored in user settings (encrypted at rest)
    \item Keys used client-side or via secure proxy with CORS
    \item Server never handles raw API keys beyond transit
\end{itemize}

\subsection{Note Storage Architecture}

\textbf{Decision:} Notes stored in S3 as JSON files, not in PostgreSQL BLOB columns.

\textbf{Rationale:}
\begin{itemize}
    \item S3 scales independently from database
    \item Simpler backup and disaster recovery
    \item Better versioning and audit trail support
    \item Natural compatibility with PWA offline storage
\end{itemize}

\subsection{UUID v7 for All IDs}

\textbf{Decision:} All primary keys use UUID v7 (not auto-increment integers).

\textbf{Rationale:}
\begin{itemize}
    \item Cryptographically secure (no ID enumeration attacks)
    \item Sortable by timestamp (useful for time-series queries)
    \item No central coordination needed for distributed systems
    \item Zero collision risk
\end{itemize}

\subsection{Semantic Search First}

\textbf{Decision:} Semantic vector search is the primary search mode; fuzzy search is fallback.

\textbf{Rationale:}
\begin{itemize}
    \item Better user experience (understands meaning, not just keywords)
    \item Integrates naturally with RAG chat
    \item Single-mode search UI (less cognitive load)
\end{itemize}

\subsection{No Collaborative Editing}

\textbf{Decision:} Notes are single-user only (no real-time collaboration).

\textbf{Rationale:}
\begin{itemize}
    \item Simplifies architecture (no WebSockets, CRDT, conflict resolution)
    \item Reduces scope and time-to-market

\end{itemize}

\section{Glossary}
%==============================================================================

\begin{description}
    \item[RAG] Retrieval-Augmented Generation. Technique for answering questions by retrieving relevant documents and using an LLM to synthesize answers.
    \item[UUID v7] Universally Unique Identifier version 7. Cryptographically secure, sortable by timestamp.
    \item[pgvector] PostgreSQL extension for storing and querying vector embeddings.
    \item[Semantic Search] Search based on meaning using vector embeddings, not keyword matching.
    \item[SM-2] Spaced Repetition algorithm for optimal flashcard review scheduling.
    \item[PWA] Progressive Web App. Web application that works offline and can be installed.
    \item[JWT] JSON Web Token. Stateless authentication mechanism.
    \item[S3] AWS Simple Storage Service for object storage.
\end{description}
